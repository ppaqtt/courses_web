入门介绍与环境搭建

📌 知识点讲解

C# 是由微软公司开发的一种现代、通用、面向对象的编程语言。它是 .NET 框架的主要编程语言，广泛应用于 Windows 应用开发、Web 开发、移动应用开发和游戏开发。

C# 的特点：

  • 简单易学，语法与 C/C++ 相似
  • 完全面向对象
  • 类型安全
  • 自动内存管理（垃圾回收）
  • 跨平台支持（.NET Core/.NET 5+）

💻 环境搭建

要开始学习 C#，需要安装以下工具：

  1. Visual Studio 或 Visual Studio Code
  2. .NET SDK（运行时和开发工具）
bash 
# 检查 .NET 版本
dotnet --version

# 创建新项目
dotnet new console -n MyFirstApp

# 进入项目目录
cd MyFirstApp

# 运行项目
dotnet run

    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Hello, World!");
            Console.WriteLine("欢迎学习 C#！");
        }
    }
}

基本语法

📌 数据类型

C# 是强类型语言，每个变量都必须声明其类型。C# 支持以下数据类型：

  • 值类型：int, double, bool, char, decimal 等
  • 引用类型：string, object, class 等
C# 
using System;

class Program
{
    static void Main()
    {
        // 值类型
        int age = 25;
        double height = 1.75;
        bool isStudent = true;
        char grade = 'A';
        decimal price = 99.99m;
        
        // 引用类型
        string name = "张三";
        
        Console.WriteLine($"姓名: {name}");
        Console.WriteLine($"年龄: {age}");
        Console.WriteLine($"身高: {height}m");
    }
}

📌 跳转语句

C# 
using System;

class Program
{
    static void Main()
    {
        // break - 跳出循环
        for (int i = 1; i <= 10; i++)
        {
            if (i == 5) break;
            Console.WriteLine($"break: {i}");
        }
        
        // continue - 跳过当前迭代
        Console.WriteLine("\ncontinue 示例：");
        for (int i = 1; i <= 5; i++)
        {
            if (i == 3) continue;
            Console.WriteLine($"continue: {i}");
        }
        
        // return - 退出方法
        Console.WriteLine("这行会执行");
        return;
        Console.WriteLine("这行不会执行");
    }
}

❌ 错误示例

常见错误：无限循环
// 错误：条件永远为真，导致无限循环
int i = 0;
while (i < 10)
{
    Console.WriteLine(i);
    // 忘记 i++，导致无限循环！
}

// 正确做法：
int i = 0;
while (i < 10)
{
    Console.WriteLine(i);
    i++; // 必须有迭代更新
}

📌 Dictionary

using System;
using System.Collections.Generic;

class Program
{

    {

using System;

class Person
{

using System;

class Calculator
{

        return a + b;

    {
        return a + b;
    }

    {
        return a + b;
    }
}

}

using System;

class Counter
{

}

继承与接口

📌 继承

C# 
using System;

// 基类（父类）
class Animal
{
    protected string name;
    
    public Animal(string name)
    {
        this.name = name;
    }
    
    public virtual void Eat()
    {
        Console.WriteLine($"{name} 在吃东西");
    }
    
    public void Sleep()
    {
        Console.WriteLine($"{name} 在睡觉");
    }
}

// 派生类（子类）
class Dog : Animal
{
    public Dog(string name) : base(name) { }
    
    // 重写父类方法
    public override void Eat()
    {
        Console.WriteLine($"{name} 在啃骨头");
    }
    
    // 新增方法
    public void Bark()
    {
        Console.WriteLine($"{name} 在汪汪叫");
    }
}

class Cat : Animal
{
    public Cat(string name) : base(name) { }
    
    public override void Eat()
    {
        Console.WriteLine($"{name} 在吃鱼");
    }
    
    public void Meow()
    {
        Console.WriteLine($"{name} 在喵喵叫");
    }
}

class Program
{
    static void Main()
    {
        Dog dog = new Dog("旺财");
        dog.Eat();   // 调用重写的方法
        dog.Bark();  // 调用子类特有的方法
        dog.Sleep(); // 调用父类方法
        
        Cat cat = new Cat("咪咪");
        cat.Eat();
        cat.Meow();
        
        // 多态
        Animal animal = new Dog("小黑");
        animal.Eat(); // 调用 Dog 的 Eat 方法
    }
}

📌 接口

using System;

// 定义接口
interface IFlyable
{
    void Fly();
    int MaxAltitude { get; }
}

interface ISwimmable
{
    void Swim();
    double MaxDepth { get; }

// 实现接口
class Bird : IFlyable

    {
        Name = name;
    }
    
    public void Fly()
    {
        Console.WriteLine($"{Name} 在飞翔");
    }
    
    public int MaxAltitude => 10000;
}

class Duck : IFlyable, ISwimmable
{
    public string Name { get; set; }
    
    public Duck(string name)
    {
        Name = name;
    }
    
    public void Fly()
    {
        Console.WriteLine($"{Name} 在低空飞行");
    }
    
    public int MaxAltitude => 500;
    
    public void Swim()
    {
        Console.WriteLine($"{Name} 在游泳");
    }
    
    public double MaxDepth => 2.0;
}

class Program
{
    static void Main()
    {
        Bird eagle = new Bird("老鹰");
        eagle.Fly();

📌 abstract 抽象类

C# 
using System;

// 抽象类
abstract class Shape
{
    public abstract double Area { get; }
    public abstract double Perimeter { get; }
    
    public void DisplayInfo()
    {
        Console.WriteLine($"面积: {Area}, 周长: {Perimeter}");
    }
}

class Circle : Shape
{
    public double Radius { get; set; }
    
    public Circle(double radius)
    {
        Radius = radius;
    }
    
    public override double Area => Math.PI * Radius * Radius;
    public override double Perimeter => 2 * Math.PI * Radius;
}

class Rectangle : Shape
{
    public double Width { get; set; }
    public double Height { get; set; }
    
    public Rectangle(double width, double height)
    {
        Width = width;
        Height = height;
    }
    
    public override double Area => Width * Height;
    public override double Perimeter => 2 * (Width + Height);
}

class Program
{
    static void Main()
    {
        Shape circle = new Circle(5);
        circle.DisplayInfo();
        
        Shape rectangle = new Rectangle(4, 6);
        rectangle.DisplayInfo();
    }
}

❌ 错误示例

常见错误：多重继承
// 错误：C# 不支持类的多重继承
class A { }
class B { }
class C : A, B { } // ❌ 编译错误！

interface IA { }

📝 练习题

1. 类和接口的主要区别是什么？
参考答案：

类可以包含实现（方法体）和状态（字段），只能单继承；接口只定义方法签名，不包含实现，可以多实现。

2. virtual 和 abstract 的区别是什么？
参考答案：

virtual 方法有默认实现，可以被子类重写；abstract 方法没有实现，必须在子类中重写，且只能在抽象类中定义。

异常处理

📌 try-catch 语句

C# 
using System;

class Program
{
    static void Main()
    {
        try
        {
            // 可能抛出异常的代码

            Console.WriteLine(numbers[5]); // 索引越界

        {
            Console.WriteLine($"其他错误: {ex.Message}");

        finally
        {
            Console.WriteLine("无论是否发生异常，都会执行");
        }
    }
}

📌 自定义异常

C# 
using System;

// 自定义异常
class InvalidAgeException : Exception
{
    public InvalidAgeException(string message) : base(message) { }
    
    public InvalidAgeException(string message, Exception inner) 
        : base(message, inner) { }
}

class Person
{
    private int age;
    
    public int Age
    {
        get { return age; }
        set
        {
            if (value < 0 || value > 150)
                throw new InvalidAgeException($"年龄 {value} 无效");
            age = value;
        }
    }
}

class Program
{
    static void Main()
    {
        try
        {
            Person p = new Person();
            p.Age = 200; // 抛出自定义异常
        }

💻 完整代码

C# 
using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

class Student
{
    public int Id { get; set; }
    public string Name { get; set; }
    public int Age { get; set; }
    public string Gender { get; set; }
    public double Score { get; set; }
    
    public override string ToString()
    {
        return $"ID: {Id}, 姓名: {Name}, 年龄: {Age}, 性别: {Gender}, 成绩: {Score}";
    }
}

class StudentManager
{
    private List students = new List();
    private const string filePath = "students.txt";
    private int nextId = 1;
    
    public StudentManager()
    {
        LoadStudents();
    }
    
    public void AddStudent(Student student)
    {
        student.Id = nextId++;
        students.Add(student);
        SaveStudents();
        Console.WriteLine("学生添加成功！");
    }
    
    public Student GetStudentById(int id)
    {
        return students.FirstOrDefault(s => s.Id == id);
    }
    
    public List SearchStudents(string keyword)
    {
        return students.Where(s => 
            s.Name.Contains(keyword) || 
            s.Gender.Contains(keyword)
        ).ToList();
    }
    
    public bool UpdateStudent(int id, Student updatedStudent)
    {
        var student = GetStudentById(id);
        if (student != null)
        {
            student.Name = updatedStudent.Name;
            student.Age = updatedStudent.Age;
            student.Gender = updatedStudent.Gender;
            student.Score = updatedStudent.Score;
            SaveStudents();
            return true;
        }
        return false;
    }
    
    public bool DeleteStudent(int id)
    {
        var student = GetStudentById(id);
        if (student != null)
        {
            students.Remove(student);
            SaveStudents();
            return true;
        }
        return false;
    }
    
    public void ShowAllStudents()
    {
        if (students.Count == 0)
        {
            Console.WriteLine("暂无学生信息");
            return;
        }
        foreach (var student in students)
        {
            Console.WriteLine(student);
        }
    }
    
    public void ShowStatistics()
    {
        if (students.Count == 0) return;
        
        Console.WriteLine($"\n统计信息：");
        Console.WriteLine($"总人数: {students.Count}");
        Console.WriteLine($"平均成绩: {students.Average(s => s.Score):F2}");
        Console.WriteLine($"最高成绩: {students.Max(s => s.Score)}");
        Console.WriteLine($"最低成绩: {students.Min(s => s.Score)}");
        
        var genderGroups = students.GroupBy(s => s.Gender);
        foreach (var group in genderGroups)
        {
            Console.WriteLine($"{group.Key}: {group.Count()}人");
        }
    }
    
    private void SaveStudents()
    {
        using (StreamWriter writer = new StreamWriter(filePath))
        {
            writer.WriteLine(nextId);
            foreach (var student in students)
            {
                writer.WriteLine($"{student.Id},{student.Name},{student.Age},{student.Gender},{student.Score}");
            }
        }
    }
    
    private void LoadStudents()
    {
        if (!File.Exists(filePath)) return;
        
        using (StreamReader reader = new StreamReader(filePath))
        {
            string line;
            if ((line = reader.ReadLine()) != null)
            {
                nextId = int.Parse(line);
            }
            
            while ((line = reader.ReadLine()) != null)
            {
                string[] parts = line.Split(',');
                students.Add(new Student
                {
                    Id = int.Parse(parts[0]),
                    Name = parts[1],
                    Age = int.Parse(parts[2]),
                    Gender = parts[3],
                    Score = double.Parse(parts[4])
                });
            }
        }
    }
}
