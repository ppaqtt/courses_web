    🔀 条件语句

    条件语句用于根据不同条件执行不同的代码块。Python使用 if、elif、else 关键字。

    python 
    # 简单条件判断

    grade = "A"
elif score >= 80:
    grade = "B"
elif score >= 70:
    grade = "C"
else:
    grade = "D"

# 嵌套条件
is_student = True
has_id = True
if is_student:
    if has_id:
        print("可以进入图书馆")
    else:
        print("请出示学生证")

    print("校外人员禁止入内")

    🔄 循环语句

    循环语句用于重复执行一段代码。Python提供 for 循环和 while 循环。

    python 
    # for 循环
for i in range(5):
    print(i)  # 输出: 0 1 2 3 4

# 遍历列表
fruits = ["apple", "banana", "cherry"]
for fruit in fruits:
    print(fruit)

# while 循环
count = 0
while count < 5:
    print(count)
    count += 1

# break 和 continue
for i in range(10):
    if i == 5:
        break  # 跳出循环
    print(i)

for i in range(10):
    if i % 2 == 0:
        continue  # 跳过本次循环
    print(i)

    ❌ 常见错误

    缩进错误：Python使用缩进来区分代码块，if、for、while 后的代码必须缩进。

    无限循环：while True: 如果没有 break 语句会导致无限循环。

    5. 函数定义

    📝 函数基础

    函数是可重用的代码块，用于执行特定任务。使用 def 关键字定义函数。

    python 
    # 定义函数
def greet(name):
    """向用户打招呼"""
    print(f"Hello, {name}!")

# 调用函数
greet("Alice")

# 返回值
def add(a, b):
    return a + b

result = add(3, 5)
print(result)  # 输出: 8

# 默认参数
def greet(name, greeting="Hello"):
    print(f"{greeting}, {name}!")

greet("Bob")              # 使用默认问候语
greet("Charlie", "Hi")    # 使用自定义问候语

# 可变参数
def sum_numbers(*args):
    total = 0
    for num in args:
        total += num
    return total

print(sum_numbers(1, 2, 3, 4))  # 输出: 10

# 关键字参数
def print_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_info(name="Alice", age=25, city="New York")

    🔄 匿名函数 (Lambda)

    Lambda函数是一种简洁的匿名函数，通常用于简单的操作。

    python 
    # Lambda函数
add = lambda x, y: x + y
print(add(3, 5))  # 输出: 8

# 与map、filter配合使用
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, numbers))
print(squared)  # 输出: [1, 4, 9, 16, 25]

even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
print(even_numbers)  # 输出: [2, 4]

# 排序时使用
people = [("Alice", 25), ("Bob", 20), ("Charlie", 30)]
people.sort(key=lambda x: x[1])
print(people)

    ❌ 常见错误

    参数不匹配：调用函数时传入的参数数量与定义不一致。

    忘记return：函数没有return语句时返回None。

    7. 面向对象

    🏗️ 类与对象

    面向对象编程是一种编程范式，使用类和对象来组织代码。类是对象的模板，对象是类的实例。

    python 
    # 定义类
class Person:
    # 构造方法
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    # 实例方法
    def greet(self):
        print(f"Hello, my name is {self.name}")
    
    # 获取年龄
    def get_age(self):
        return self.age

# 创建对象
person1 = Person("Alice", 25)
person2 = Person("Bob", 30)

# 调用方法
person1.greet()
print(person2.get_age())

# 修改属性
person1.age = 26
print(person1.age)

    🔗 继承

    继承允许一个类（子类）继承另一个类（父类）的属性和方法。

    python 
    # 父类
class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        print(f"{self.name} makes a sound")

# 子类继承父类
class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name)  # 调用父类构造方法
        self.breed = breed
    
    # 重写方法
    def speak(self):
        print(f"{self.name} barks: Woof!")
    
    # 新增方法
    def fetch(self):
        print(f"{self.name} fetches the ball")

# 使用子类
dog = Dog("Buddy", "Golden Retriever")
dog.speak()   # 输出: Buddy barks: Woof!
dog.fetch()   # 输出: Buddy fetches the ball

    🔒 封装、多态

    • 封装：隐藏对象内部细节，通过方法访问
    • 多态：不同对象对同一方法有不同的实现
    python 
    # 封装（私有属性）
class BankAccount:
    def __init__(self, balance):
        self.__balance = balance  # 私有属性
    
    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount
    
    def withdraw(self, amount):
        if amount > 0 and amount <= self.__balance:
            self.__balance -= amount
    
    def get_balance(self):
        return self.__balance

account = BankAccount(1000)
account.deposit(500)
print(account.get_balance())  # 输出: 1500
# print(account.__balance)  # 错误：私有属性不能直接访问

# 多态示例
class Cat(Animal):
    def speak(self):
        print(f"{self.name} meows: Meow!")

def make_sound(animal):
    animal.speak()

dog = Dog("Buddy", "Labrador")
cat = Cat("Whiskers")
make_sound(dog)   # 输出: Buddy barks: Woof!
make_sound(cat)   # 输出: Whiskers meows: Meow!

    ❌ 常见错误

    忘记self：实例方法必须以self作为第一个参数。

    访问私有属性：私有属性（以__开头）不能直接从外部访问。

    9. 错误处理

    🛡️ try-except语句

    错误处理是编程中非常重要的部分，可以防止程序因异常而崩溃。

    python 
    # 基本错误处理
try:
    num = int(input("请输入一个数字: "))
    print(f"输入的数字是: {num}")
except ValueError:
    print("错误: 请输入有效的数字")

# 多个except
try:
    file = open("example.txt", "r")
    content = file.read()
    print(content)
except FileNotFoundError:
    print("错误: 文件不存在")
except PermissionError:
    print("错误: 没有读取权限")

# 捕获所有异常
try:
    result = 10 / 0
except Exception as e:
    print(f"发生错误: {e}")

# else和finally
try:
    num = int(input("请输入数字: "))
except ValueError:
    print("输入错误")
else:
    print(f"你输入了: {num}")
finally:
    print("操作完成")

    📢 抛出异常

    使用raise语句可以主动抛出异常。

    python 
    # 抛出异常
def check_age(age):
    if age < 0:
        raise ValueError("年龄不能为负数")
    if age > 120:
        raise ValueError("年龄不合理")
    return age

try:
    check_age(-5)
except ValueError as e:
    print(f"错误: {e}")

# 自定义异常
class InvalidEmailError(Exception):
    def __init__(self, message):
        super().__init__(message)

def validate_email(email):
    if "@" not in email:
        raise InvalidEmailError(f"无效的邮箱: {email}")
    return True

try:
    validate_email("invalid-email")
except InvalidEmailError as e:
    print(e)

    ❌ 常见错误

    过度捕获异常：使用 except Exception 会捕获所有异常，包括系统退出等，应尽量捕获特定异常。

    忽略异常：空的 except 块会隐藏错误，不利于调试。

    10. 综合实战

    🎯 学生管理系统

    现在我们来创建一个完整的学生管理系统，综合运用前面学到的所有知识。

    python 
    import json

class Student:
    def __init__(self, name, age, student_id, grade):
        self.name = name
        self.age = age
        self.student_id = student_id
        self.grade = grade
    
    def to_dict(self):
        return {
            "name": self.name,
            "age": self.age,
            "student_id": self.student_id,
            "grade": self.grade
        }

class StudentManager:
    def __init__(self):
        self.students = []
        self.load_students()
    
    def load_students(self):
        try:
            with open("students.json", "r") as f:
                data = json.load(f)
                self.students = [Student(**s) for s in data]
        except FileNotFoundError:
            self.students = []
    
    def save_students(self):
        with open("students.json", "w") as f:
            data = [s.to_dict() for s in self.students]
            json.dump(data, f, indent=2)
    
    def add_student(self, student):
        if any(s.student_id == student.student_id for s in self.students):
            raise ValueError("学号已存在")
        self.students.append(student)
        self.save_students()
    
    def remove_student(self, student_id):
        self.students = [s for s in self.students if s.student_id != student_id]
        self.save_students()
    
    def search_student(self, keyword):
        return [s for s in self.students if keyword in s.name or keyword == s.student_id]
    
    def get_all_students(self):
        return self.students
    
    def get_student_by_id(self, student_id):
        return next((s for s in self.students if s.student_id == student_id), None)

def add_student_menu(manager):
    try:
        name = input("请输入姓名: ")
        age = int(input("请输入年龄: "))
        student_id = input("请输入学号: ")
        grade = input("请输入成绩: ")
        
        student = Student(name, age, student_id, grade)
        manager.add_student(student)
        print("学生添加成功！")
    except ValueError as e:
        print(f"错误: {e}")

def search_student_menu(manager):
    keyword = input("请输入搜索关键词（姓名或学号）: ")
    results = manager.search_student(keyword)
    
    if results:
        print("\n搜索结果:")
        for s in results:
            print(f"学号: {s.student_id}, 姓名: {s.name}, 年龄: {s.age}, 成绩: {s.grade}")
    else:
        print("未找到匹配的学生")

def show_all_students(manager):
    students = manager.get_all_students()
    
    if students:
        print("\n所有学生:")
        for s in students:
            print(f"学号: {s.student_id}, 姓名: {s.name}, 年龄: {s.age}, 成绩: {s.grade}")
    else:
        print("暂无学生记录")

def main():
    manager = StudentManager()
    
    while True:
        display_menu()
        choice = input("请选择操作: ")
        
        if choice == "1":
            add_student_menu(manager)
        elif choice == "2":
            student_id = input("请输入要删除的学号: ")
            manager.remove_student(student_id)
            print("删除成功")
        elif choice == "3":
            search_student_menu(manager)
        elif choice == "4":
            show_all_students(manager)
        elif choice == "5":
            print("退出系统")
            break
        else:
            print("无效选择，请重新输入")

if __name__ == "__main__":
    main()
