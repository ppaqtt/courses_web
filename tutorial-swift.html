    1. 入门介绍与环境搭建

    Swift是Apple公司于2014年推出的一种现代编程语言，用于开发iOS、macOS、watchOS和tvOS应用。它结合了C和Objective-C的优点，同时避免了它们的缺点，是一种类型安全、快速且易于学习的语言。

    🌟 Swift的特点

    • 安全：类型安全，自动内存管理，避免空指针
    • 快速：性能接近C语言
    • 简洁：语法简洁，表达力强
    • 现代：支持泛型、闭包、协议等现代特性
    • 互操作性：与Objective-C无缝兼容

    📦 开发环境搭建

    要开发Swift应用，你需要安装Xcode，这是Apple官方的集成开发环境。

    1. 打开Mac App Store，搜索并安装Xcode
    2. 打开Xcode，同意许可协议
    3. 安装Command Line Tools（可选）
    bash 
    # 检查Swift版本

# 打开REPL（交互式环境）

    🚀 第一个Swift程序

    打开Xcode，创建一个新的Command Line Tool项目，或者在终端中创建一个 hello.swift 文件：

    swift 
    // 这是一个注释
print("Hello, World!")

    在终端运行：

    bash

    使用中文引号：print("Hello“) 会导致语法错误，必须使用英文引号。

    🔄 循环语句

    Swift提供 for-in、while 和 repeat-while 循环。

    swift 
    // for-in循环（遍历范围）

    print(i)  // 0, 1, 2, 3, 4
}

for i in 1...5 {
    print(i)  // 1, 2, 3, 4, 5
}

// for-in循环（遍历数组）

    print(fruit)

    print("\(index): \(fruit)")
}

// while循环
var count = 0

    print(count)
    count += 1
}

// repeat-while循环（至少执行一次）
var x = 0
repeat {
    print(x)
    x += 1
} while x < 5

// break和continue
for i in 0..<10 {
    if i == 5 {
        break  // 跳出循环
    }
    print(i)
}

for i in 0..<10 {
    if i % 2 == 0 {
        continue  // 跳过本次循环
    }
    print(i)
}

    7. 闭包

    📦 闭包基础

    闭包是自包含的功能代码块，可以作为参数传递或存储。闭包可以捕获和存储其所在上下文中的常量和变量。

    swift 
    // 完整形式的闭包
let add = {(a: Int, b: Int) -> Int in
    return a + b
}

print(add(3, 5))  // 输出: 8

// 简化形式（类型推断）
let multiply = { (a, b) in
    a * b
}

// 更简化（单表达式闭包自动返回）
let divide = { $0 / $1 }

// 闭包作为函数参数
func performOperation(_ operation: (Int, Int) -> Int, on a: Int, and b: Int) {
    print(operation(a, b))
}

performOperation(add, on: 10, and: 20)

// 尾随闭包（当闭包是最后一个参数时）
performOperation(on: 10, and: 20) { $0 * $1 }

// 闭包捕获值
func makeCounter() -> () -> Int {
    var count = 0
    return {
        count += 1
        return count
    }
}

let counter = makeCounter()
print(counter())  // 1
print(counter())  // 2
print(counter())  // 3

    🔗 逃逸闭包与非逃逸闭包

    swift 
    // 非逃逸闭包（默认）
func process(_ closure: () -> Void) {
    closure()
}

// 逃逸闭包（需要显式标记）
func asyncProcess(_ closure: @escaping () -> Void) {
    DispatchQueue.main.async {
        closure()
    }
}

// 自动闭包（延迟求值）
func check(_ condition: @autoclosure () -> Bool) {
    if condition() {
        print("条件为真")
    }
}

check(10 > 5)

    ❌ 常见错误

    忘记in关键字：闭包中参数列表和代码体之间必须使用 in 分隔。

    逃逸闭包未标记：当闭包需要在函数返回后执行时，必须使用 @escaping 标记。

    9. 错误处理

    ⚠️ 错误定义

    使用 enum 定义错误类型，需要遵循 Error 协议。

    swift 
    // 定义错误类型
enum ValidationError: Error {
    case invalidEmail
    case passwordTooShort(minLength: Int)
    case ageUnder18
}

// 抛出错误
func validateEmail(_ email: String) throws {
    if !email.contains("@") {
        throw ValidationError.invalidEmail
    }
}

func validatePassword(_ password: String) throws {
    if password.count < 8 {
        throw ValidationError.passwordTooShort(minLength: 8)
    }
}

// try-catch处理错误
do {
    try validateEmail("invalid-email")
    print("Email valid")
} catch ValidationError.invalidEmail {
    print("Invalid email format")
} catch {
    print("Unknown error: \(error)")
}

do {
    try validateEmail("test@example.com")
    try validatePassword("pass")
} catch ValidationError.invalidEmail {
    print("Invalid email")
} catch ValidationError.passwordTooShort(let min) {
    print("Password must be at least \(min) characters")
}

    🔄 可选错误处理

    swift 
    // try? 返回可选值
let result = try? validateEmail("test@example.com")
if result == nil {
    print("Validation failed")
}

// try! 强制解包
let success = try! validateEmail("valid@example.com")

// 自定义错误描述
extension ValidationError: LocalizedError {
    var errorDescription: String? {
        switch self {
        case .invalidEmail:
            return "Invalid email address"
        case .passwordTooShort(let min):
            return "Password must be at least \(min) characters"
        case .ageUnder18:
            return "Must be 18 or older"
        }
    }
}

    ❌ 常见错误

    未处理抛出的错误：调用可抛出函数时必须使用 try、try? 或 try!。

    错误类型未遵循Error协议：自定义错误必须遵循 Error 协议。

    🎯 项目概述

    创建一个简单的待办事项应用，包含添加、删除、完成等功能。

    swift 
    // TodoItem结构体
struct TodoItem: Identifiable {
    let id = UUID()
    var title: String
    var description: String
    var isCompleted: Bool
    var createdAt: Date
}

// TodoManager类
class TodoManager {
    private var todos: [TodoItem] = []
    
    // 添加待办
    func addTodo(title: String, description: String = "") {
        let todo = TodoItem(
            title: title,
            description: description,
            isCompleted: false,
            createdAt: Date()
        )
        todos.append(todo)
    }
    
    // 删除待办
    func deleteTodo(at index: Int) {
        guard index >= 0 && index < todos.count else { return }
        todos.remove(at: index)
    }
    
    // 切换完成状态
    func toggleCompletion(at index: Int) {
        guard index >= 0 && index < todos.count else { return }
        todos[index].isCompleted.toggle()
    }
    
    // 获取所有待办
    func getAllTodos() -> [TodoItem] {
        return todos
    }
    
    // 获取未完成的待办数量
    func getPendingCount() -> Int {
        return todos.filter { !$0.isCompleted }.count
    }
}

// 使用示例
let manager = TodoManager()
manager.addTodo(title: "学习Swift", description: "完成第10章")
manager.addTodo(title: "购物")
manager.toggleCompletion(at: 0)
print("已完成: \(manager.getPendingCount())")

    📊 扩展功能

    swift 
    // 扩展TodoManager
extension TodoManager {
    func getSortedTodos() -> [TodoItem] {
        return todos.sorted { $0.createdAt > $1.createdAt }
    }
    
    func searchTodos(_ query: String) -> [TodoItem] {
        return todos.filter {
            $0.title.lowercased().contains(query.lowercased())
        }
    }
    
    func clearCompleted() {
        todos = todos.filter { !$0.isCompleted }
    }
    
    func exportTodos() -> String {
        var result = ""
        for (index, todo) in todos.enumerated() {
            let status = todo.isCompleted ? "[✓]" : "[ ]"
            result += "\(index + 1). \(status) \(todo.title)\n"
        }
        return result
    }
}

// 使用扩展功能
let sortedTodos = manager.getSortedTodos()
let searchResults = manager.searchTodos("学习")
print(manager.exportTodos())
manager.clearCompleted()
